\subsection*{(ii)}
\[
	\begin{pmatrix}
		\cos\theta  & \sin\theta \\
		-\sin\theta & \cos\theta
	\end{pmatrix}
\]

Let $A = \begin{pmatrix}
		\cos\theta  & \sin\theta \\
		-\sin\theta & \cos\theta
	\end{pmatrix}$

\begin{align}
	\det(A) & = \cos^2\theta + \sin^2\theta = 1
\end{align}

\[
	A^{-1} = \begin{pmatrix}
		\cos\theta & -\sin\theta \\
		\sin\theta & \cos\theta
	\end{pmatrix}
\]
