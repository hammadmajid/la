	\left(\begin{array}{ccc|ccc}

\[
	\left(\begin{array}{ccc|ccc}
			1 & 1  & -2 & 5 & 0  & 0  \\
			2 & 2  & 1  & 0 & 10 & 0  \\
			2 & -8 & 1  & 0 & 0  & 10
		\end{array}\right)

\[
	\left(\begin{array}{ccc|ccc}
			1 & 1   & -2 & 5   & 0  & 0  \\
			0 & 0   & 5  & -10 & 10 & 0  \\
			0 & -10 & 5  & -10 & 0  & 10
		\end{array}\right)
\]

$R_2 \leftrightarrow R_3$:
\[
	\left(\begin{array}{ccc|ccc}
			1 & 1   & -2 & 5   & 0  & 0  \\

			0 & 0   & 5  & -10 & 10 & 0
		\end{array}\right)

\[
	\left(\begin{array}{ccc|ccc}

$R_3 \leftarrow \frac{1}{5}R_3$:
\[
	\left(\begin{array}{ccc|ccc}
