\subsection*{(i)}
\[
	\begin{pmatrix}
		\frac{1}{2}(e^x + e^{-x}) & \frac{1}{2}(e^x - e^{-x}) \\
		\frac{1}{2}(e^x - e^{-x}) & \frac{1}{2}(e^x + e^{-x})
	\end{pmatrix}
\]

Let $A = \begin{pmatrix}
		\cosh x & \sinh x \\
		\sinh x & \cosh x
	\end{pmatrix}$ where $\cosh x = \frac{1}{2}(e^x + e^{-x})$ and $\sinh x = \frac{1}{2}(e^x - e^{-x})$

\begin{align}
	\det(A) = \cosh^2 x - \sinh^2 x = 1
\end{align}

\[
	A^{-1} = \begin{pmatrix}
		\cosh x  & -\sinh x \\
		-\sinh x & \cosh x
	\end{pmatrix} = \begin{pmatrix}
		\frac{1}{2}(e^x + e^{-x})  & -\frac{1}{2}(e^x - e^{-x}) \\
		-\frac{1}{2}(e^x - e^{-x}) & \frac{1}{2}(e^x + e^{-x})
	\end{pmatrix}
\]
