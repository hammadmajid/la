	x_1 + x_2 + 2x_3   & = 8  \\
	-x_1 - 2x_2 + 3x_3 & = 1  \\
	3x_1 - 7x_2 + 4x_3 & = 10
\end{align*}

	\left(\begin{array}{ccc|c}

\[
	\left(\begin{array}{ccc|c}
			1 & 1 & 2  & 8  \\
			0 & 1 & -5 & -9 \\
			0 & 0 & 1  & 2
		\end{array}\right)
\]

Back substitution:
\begin{align*}
	x_3 &= 2 \\
	x_2 - 5x_3 &= -9 \Rightarrow x_2 = -9 + 5(2) = 1 \\
	x_1 + x_2 + 2x_3 &= 8 \Rightarrow x_1 = 8 - 1 - 2(2) = 3
\end{align*}

$\boxed{(x_1, x_2, x_3) = (3, 1, 2)}$
